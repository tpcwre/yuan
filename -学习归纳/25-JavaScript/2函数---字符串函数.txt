//-- substr(x,n) 截取从x 到 n 的子串

//-- substing(x,n)   x,n两者中的较小值作为子字符串的起始点,较大值做为终点





//-- substr(x,n) 截取从x 到 n 的子串


	var str = 'abcdefg'.substr(2,3);		
	
	alert(str);			//结果  cde
	
	




//-- substing(x,n)   x,n两者中的较小值作为子字符串的起始点,较大值做为终点


	var str2 = "abcdefg".substring(0,3);

	alert(str2);		//结果：abc

	var str3 = "abcdefg".substring(3,0);

	alert(str3);		//结果：abc
