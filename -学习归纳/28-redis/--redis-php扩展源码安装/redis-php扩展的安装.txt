
//===== 安装redis

1  上传并解压 redis 和 phpredis

2  进入 redis目录    
	make 			//编译
	make install 		//安装

3 cp ./redis.conf /etc		//将配置文件拷贝到/etc下

4 vim /etc/redis.conf
	将 daemonize no 改为 	daemonize yes  	//改成后台运行	
	将 # requirepass foobared	改  requirepass 密码		//设置密码
	登陆时需要  "auth 密码"		方式验证

5 redis-server /etc/conf	//启用/etc下的配置来后台启动redis 


6 redis-cli		//启动redis客户端进行数据库操作
  auth root1234		//使用密码登陆



//===== redis 关联 php 

1 进入phpredis目录

2 /usr/local/php/bin/phpize		//生成configure文件

3 ./configure --with-php-config=/usr/local/php/bin/php-config 	//安装phpredis关联文件

4 make 				//编译

5 make install			//安装

6 vim /usr/local/php/etc/php.ini

	添加  extension=redis.so  扩展模块


7 service php-fpm restart  	重启php
  service nginx restart		重启ngnix

  ps -aux | grep redis		重启redis
  kill -9 xxxx		
  redis-server /etc/redis.conf
