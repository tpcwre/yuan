
//----- 安装 redis

	cd /usr/local/src		//进入指定目录

	rz redis-3.0.1.tar.gz		//上传redis源码包

	tar vzxf redis-3.0.1.tar.gz	//解压
	
	cd redis-3.0.1			//进入redis安装目录


	//-- redis服务端启动方式一：前台运行
	make && make install		//编译并安装redis
	redis-server			//启动服务端
	ctrl+c				//退出服务端
	

	//-- redis服务端启动方式二：后台运行			
	vim redis.conf			//打开配置文件
		设置 daemonize yes	//设置服务端后台运行
		     requirepass root1234 //设置密码
	redis-server redis.conf		//启动服务端
	ps -aux | grep redis 		//获取redis 服务进程ID
	kill -9 id			//结束redis 服务进程



	redis-cli			//启动客户端





//----- 关联PHP

	rz redis-2.2.7.tgz			//上传关联依赖包

	tar vzxf redis-2.2.7.tgz		//解压
	
	/usr/local/php/bin/phpize 		//运行phpize 生成configure等文件

	./configure --with-php-config=/usr/local/php/bin/php-config 	//php关联处理

	make && make install 			//编译并安装 

	vim /usr/local/php/etc/php.ini 		//打开php配置文件

		extension=redis.so		//在配置文件中添加redis模块代码 

	service php-fpm restart			//刷新php内核(该刷新方式为lnmp环境)

	redis-server ../redis-3.0.1/redis.conf 	//用配置文件方式启动redis服务端
